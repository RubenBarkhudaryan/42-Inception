            - MYSQL_ROOT_PASSWORD_FILE=db_root_password
        networks:
            - inception
        ports:
            - "3306:3306"
        restart: always

    wordpress:
        build: ./requirements/wordpress
        container_name: wordpress
        volumes:
            - wordpress_data:/var/www/html
        secrets:
            - db_password
        environment:
            - WORDPRESS_DB_HOST=database:3306
            - WORDPRESS_DB_USER=wordpress
            - WORDPRESS_DB_PASSWORD_FILE=db_password
            - WORDPRESS_DB_NAME=wordpress
        networks:
            - inception
        ports:
            - "9000:9000"
        depends_on:
            - webserver
            - database
        restart: always

# Yes, when you create a custom network like "inception" in Docker, the containers that are connected to that network can communicate with each other using their service names as hostnames. This means that the webserver container can access the WordPress container using the hostname "wordpress" without needing to map ports to the host machine. However, if you want to access the WordPress site from outside the Docker network (e.g., from your web browser on your host machine), you will still need to map the appropriate ports in the docker-compose file.

# so I can just simply write expose in dockerfile and it will work ?

# Yes, you can use the EXPOSE instruction in your Dockerfile to indicate which ports the container will listen on at runtime. However, the EXPOSE instruction does not actually publish the ports to the host machine; it is more of a documentation feature that informs users of the container which ports are intended to be used. To make the application accessible from outside the Docker network, you still need to use port mapping in your docker-compose file (e.g., "443:443") to map the container's ports to the host machine's ports.

# if I decide to use ports in docker compose file anyway. how should I configure them ?

# If you decide to use port mapping in your docker-compose file, you should configure it based on the ports that your application is listening on inside the container. For example, if your WordPress application is listening on port 80 inside the container, and you want to access it from your host machine, you can map it like this:
# ```yamlports:
#     - "80:80"
# ```
# This means that port 80 on the host machine will be forwarded to port 80 in the container. If you want to use a different port on the host machine (e.g., 9000), you can configure it like this:
# ```yamlports:
#     - "9000:80"
# ```
# This means that port 9000 on the host machine will be forwarded to port 80 in the container. You can choose any available port on the host machine for the left side of the mapping, but the right side should match the port that your application is listening on inside the container. In this case, since WordPress typically listens on port 80, you would map it to port 80 in the container.
