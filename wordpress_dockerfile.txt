FROM		debian:bookworm-slim

RUN			mkdir -p /var/www/html

RUN			apt-get update && \
			apt-get upgrade -y && \
			apt-get install -y \
				php8.2 \
				php8.2-fpm \
				php8.2-mysql \
				php8.2-curl \
				php8.2-gd \
				php8.2-mbstring \
				php8.2-xml \
				php8.2-zip \
				php8.2-intl \
				php8.2-soap \
				php8.2-bcmath \
				php8.2-json \
				php8.2-cli \
				php8.2-readline \
				php8.2-ldap \
				php8.2-tideways \
				curl \
				wget \
				unzip \
			&& apt-get clean && \
			rm -rf /var/lib/apt/lists/*

RUN			sed -i "s/listen = .*/listen = 0.0.0.0:9000/" /etc/php/8.2/fpm/pool.d/www.conf

RUN			curl -O https://raw.githubusercontent.com/wp-cli/builds/gh-pages/phar/wp-cli.phar \
			&& chmod +x wp-cli.phar \
			&& mv wp-cli.phar /usr/local/bin/wp

COPY		./tools/wordpress.sh /wordpress.sh

COPY		./conf/www.conf /etc/php/8.2/fpm/pool.d/www.conf

RUN			chmod +x /wordpress.sh

EXPOSE		9000

ENTRYPOINT	["/wordpress.sh"]

CMD			["php-fpm8.2", "-F"]

# can you explain what that sed is doing 

# The sed command is used to modify the PHP-FPM configuration file (www.conf) to change the listen directive. By default, PHP-FPM listens on a Unix socket, but in this case, we want it to listen on all network interfaces on port 9000. The sed command searches for the line that starts with "listen = " and replaces it with "listen = 0.0.0.0:9000". This allows the PHP-FPM service to accept incoming connections from any IP address on port 9000, which is necessary for the WordPress application to communicate with the PHP-FPM service.

# explain the string given to sed

#The string given to sed is a substitution command that follows the format "s/pattern/replacement/". In this case, the pattern is "listen = .*", which matches any line that starts with "listen = " followed by any characters (indicated by ".*"). The replacement is "listen = 0.0.0.0:9000"
