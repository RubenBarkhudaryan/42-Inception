#!/bin/bash
mkdir -p /var/www/html
cd /var/www/html

rm -rf *

# Download and set up WordPress using WP-CLI
if wp core download --allow-root; then
    echo "WordPress core downloaded successfully."
else
    echo "Failed to download WordPress core."
    exit 1
fi

if [ ! -f wp-config.php ]; then
    if cp wp-config-sample.php wp-config.php; then
        echo "wp-config.php created."
    else
        echo "Failed to create wp-config.php."
        exit 1
    fi
fi

if sed -i "s/define( 'DB_NAME', '.*' );/define( 'DB_NAME', '${MYSQL_DATABASE}' );/g" wp-config.php && \
   sed -i "s/define( 'DB_USER', '.*' );/define( 'DB_USER', '${MYSQL_USER}' );/g" wp-config.php && \
   sed -i "s/define( 'DB_PASSWORD', '.*' );/define( 'DB_PASSWORD', '${MYSQL_PASSWORD}' );/g" wp-config.php && \
   sed -i "s/define( 'DB_HOST', '.*' );/define( 'DB_HOST', '${MYSQL_HOSTNAME}' );/g" wp-config.php; then
    echo "Database credentials updated in wp-config.php."
else
    echo "Failed to update database credentials in wp-config.php."
    exit 1
fi

if wp core install --url="$DOMAIN_NAME" \
    --title="$WORDPRESS_TITLE" \
    --admin_user="$WORDPRESS_ROOT_USERNAME" \
    --admin_password="$WORDPRESS_ROOT_PASSWORD" \
    --admin_email="$WORDPRESS_ROOT_EMAIL" \
    --skip-email --allow-root; then
    echo "WordPress installed successfully."
else
    echo "Failed to install WordPress."
    exit 1
fi

if wp user create "$WORDPRESS_USER_USERNAME" "$WORDPRESS_USER_EMAIL" \
    --role=subscriber \
    --user_pass="$WORDPRESS_USER_PASSWORD" \
    --allow-root; then
    echo "WordPress user created successfully."
else
    echo "Failed to create WordPress user."
    exit 1
fi

chown -R www-data:www-data /var/www/html
chmod -R 755 /var/www/html

if wp core update --allow-root; then
    echo "WordPress core updated successfully."
else
    echo "Failed to update WordPress core."
    exit 1
fi

mkdir -p /run/php
chown -R www-data:www-data /run/php

if wp theme install twentyfourteen --activate --allow-root; then
    echo "WordPress theme installed and activated successfully."
else
    echo "Failed to install and activate WordPress theme."
    exit 1
fi

exec "$@";

# so now the script is ready ?

# Yes, the script is now ready to set up WordPress with the specified configurations and handle any potential errors during the process. It will download WordPress, configure the database connection, install WordPress, create a user, set permissions, update WordPress core, and install a theme. Finally, it will execute any additional commands passed to the script.

# where I should pass the environment variables like MYSQL_DATABASE, WORDPRESS_TITLE, etc. ?
# can I do it in the docker-compose.yml file ?

# Yes, you can pass the environment variables in the docker-compose.yml file. You can define them under the service that runs this script. For example:

  wordpress:
    build: .
    environment:
      - MYSQL_DATABASE=your_database_name
      - MYSQL_USER=your_database_user
      - MYSQL_PASSWORD=your_database_password
      - MYSQL_HOSTNAME=your_database_host
      - DOMAIN_NAME=your_domain_name
      - WORDPRESS_TITLE=your_wordpress_title
      - WORDPRESS_ROOT_USERNAME=your_admin_username
      - WORDPRESS_ROOT_PASSWORD=your_admin_password
      - WORDPRESS_ROOT_EMAIL=your_admin_email
      - WORDPRESS_USER_USERNAME=your_user_username
      - WORDPRESS_USER_EMAIL=your_user_email
      - WORDPRESS_USER_PASSWORD=your_user_password
